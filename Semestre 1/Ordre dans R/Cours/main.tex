\mysection{Ordre dans l'ensemble $\mathbb{R}$}
\begin{Définition}
Soient $a$ et $b$ deux nombres réels.
\begin{enumerate}
	\item On dit que $a$ est \textcolor{red}{inférieur ou égal} à $b$, si $b - a\geq 0$, et on écrit $a\leq b$ ou $b\geq a$.
	\item On dit que $a$ est \textcolor{red}{inférieur strictement} à $b$, si $b - a > 0$, et on écrit $a > b$ ou $b > a$.
\end{enumerate}
\end{Définition}

\textcolor{red}{Remarque :}
\begin{itemize}
	\item $a\leq b$ signifie que $a < b$ ou $a = b$.
	\item Si $a < b$ alors $a\leq b$. (le contraire est faux).
\end{itemize}

\begin{Exemple}
\begin{enumerate}
	\item On a $\dfrac{3}{7} < \dfrac{5}{8}$ car : $\dfrac{5}{8} - \dfrac{3}{7} > 0$.
	\item Soit $n\in\mathbb{N}$. On a $\dfrac{2n-1}{2n} < \dfrac{2n}{2n + 1}$.
\end{enumerate}
\end{Exemple}

\begin{Application}
Soient $a$ et $b$ deux réels strictement positifs.
\begin{enumerate}
	\item Comparer $a^2 + b^2$ et $2ab$.
	\item En déduire que $a^2 + \dfrac{1}{a^2} \geq 2$.
\end{enumerate}
\end{Application}

\mysection{Ordre et opérations}

\mysubsection{1}{Ordre et addition}
\begin{Proposition}
Soient $a,b,c$ et $d$ des nombres réels.
\begin{itemize}
	\item Si $a\leq b$, alors $a + c \leq b + c$.
	\item Si $(a \leq b$ et $c\leq d)$, alors $a+c \leq b+d$.
\end{itemize}
\end{Proposition}

\begin{Exemple}
\begin{itemize}
	\item Soient $x$ et $y$ deux nombres réels tels que $x\leq 3$ et $y\leq -10$.\\
Alors $x + y \leq 3 + (-10)$ c'est-à-dire $x + y \leq -7$.
	\item Soit $a$ un nombre réel tel que $a + 5 \leq 6$, donc $a + 5 + (-5) \leq 6 + (-5)$ c'est-à-dire $a\leq 1$.
\end{itemize}
\end{Exemple}

\begin{Application}
Soient $x$ et $y$ deux nombres réels tels que $x\leq -\dfrac{5}{3}$ et $y\leq \dfrac{7}{4}$.\\
Montrer que $x + y\leq \dfrac{1}{12}$.
\end{Application}
\textcolor{red}{Remarque :}
\begin{itemize}
	\item Si $a\leq b$ et $c < d$, alors $a + c < b + d$.
\end{itemize}

\begin{itemize}
	\item Si $a\geq 0$ et $b\geq 0$ et $a + b = 0$, alors $(a = 0$ et $b = 0)$.
	\item Si $a\leq b$ et $b \leq c$, alors $a \leq c$.
\end{itemize}

\mysubsection{2}{Ordre et multiplication}
\begin{Proposition}
Soient $a,b,c$ et $d$ des nombres réels.
\begin{itemize}
	\item Si $(a\leq b$ et $c\geq 0)$, alors $ac \leq bc$.
	\item Si $(a\leq b$ et $c\leq 0)$, alors $ac \geq bc$.
	\item Si $(ac\leq bc$ et $c > 0)$, alors $a \leq b$.
	\item Si $(ac\leq bc$ et $c < 0)$, alors $a \geq b$.
	\item Si $(0 \leq a\leq b$ et $ 0 \leq c\leq d)$, alors $ac \leq bd$.
\end{itemize}
\end{Proposition}

\begin{Exemple}

\begin{itemize}
	\item Soit $x \leq \dfrac{1}{2}$, alors $2\times x \leq 2\times \dfrac{1}{2}$ (car $2 > 0)$ c'est-à-dire $2x\leq 1$.
	\item Soit $x < \dfrac{1}{2}$, alors $(-2) \times x > (-2) \times \dfrac{1}{2}$  (car $ -2 < 0$ ) c'est-à-dire $-2x > -1$.
	\item Soit $3x < 5$, alors $\left(\dfrac{1}{3}\right)\times (3x) < \left(\dfrac{1}{3}\right)\times 5$ (car $\dfrac{1}{3} > 0$) c'est-à-dire $x < \dfrac{5}{3}$.
	\item Soit $0\leq x\leq 4,5$ et $0\leq y\leq 2$, alors $xy \leq 4,5\times 2$ c'est-à-dire $xy \leq 11$.
\end{itemize}
\end{Exemple}

\begin{Application}
	Soit $0\leq 2x\leq 3$ et $0\leq 5y\leq 75$. Montrer que $xy\leq \dfrac{45}{2}$.
\end{Application}


\mysubsection{3}{Ordre et inverse}
\begin{Proposition}
Soient $a$ et $b$ deux nombres réels.
\begin{itemize}
	\item $ 0 < a \leq b$ signifie que $0 < \dfrac{1}{b} \leq \dfrac{1}{a}$.
	\item $a\leq b < 0$ signifie que $\dfrac{1}{b} \leq \dfrac{1}{a} < 0$.
\end{itemize}
\end{Proposition}

\begin{Exemple}
\begin{itemize}
	\item $0 < x \leq 5$, donc $\dfrac{1}{5} \leq \dfrac{1}{x}$.
	\item $-\dfrac{2}{3}\leq x<0$, donc $\dfrac{1}{x} \leq \dfrac{1}{-\dfrac{2}{3}}$ c'est-à-dire $\dfrac{1}{x}\leq -\dfrac{3}{2}$.
\end{itemize}
\end{Exemple}

\begin{Application}
Soit $x$ un nombre réel tel que : $0 < \dfrac{1}{2x + 1} < \dfrac{3}{2}$. Montrer que $-\dfrac{1}{6} < x$.
\end{Application}

\mysubsection{3}{Valeur absolue et propriétés}
\begin{Activité}
On considère les points $A,B,C,D,E$ et $F$ représentés sur $(\Delta)$ ci-dessous :
\begin{center}
\begin{tikzpicture} 
\draw[-] (-7,0) -- (4,0); % Dessiner la droite 
\foreach \x in {-6,-5,-4,-3,-2,-1,0,1,2,3} % Dessiner les graduations
 \draw (\x,0.1) -- (\x,-0.1); 
 \node at (-6,0.3) {F}; 
 \node at (-5,0.3) {D}; 
 \node at (-3,0.3) {A};
 \node at (-1,0.3) {C}; 
 \node at (0,0.3) {O}; 
 \node at (2,0.3) {E}; 
 \node at (3,0.3) {B}; 
 \foreach \x in {-6,-5,-4,-3,-2,-1,0,1,2,3} % Étiqueter les graduations
 \node at (\x,-0.3) {\x};
 \end{tikzpicture}
\end{center}
\begin{enumerate}
	\item Déterminer les distances $OB$, $OC$, $OD$, $OE$ et $OF$.
\end{enumerate}
\end{Activité}

\begin{Définition}
Sur un axe normé, $x$ est l'abscisse d'un point $M$. \textcolor{red}{La valeur absolue} de $x$ est la distance entre l'origine du repère et le point $M$; on la note $|x|$.

En d'autre termes : $OM = |x|$ (où $O$ est l'origine du repère).
\end{Définition}
\begin{Exemple}
\begin{itemize}
	\item $|4| = 4$ et $|-3| = 3$ et $|-\dfrac{1}{2}| = \dfrac{1}{2}$.
	\item $|1 - \sqrt{2}| = \sqrt{2} - 1$.
\end{itemize}
\end{Exemple}

\begin{Proposition}
Soit $x$ un nombre réel. On a 
$|x| = 
\begin{cases} 
x \text{ si } x\geq 0 \\
-x \text{ si } x\leq 0
\end{cases}
$
\end{Proposition}

\begin{Définition}
Si $a$ et $b$ sont les abscisses respectives de deux points $A$ et $B$ sur un axe normé (droite graduée), alors la distance entre $a$ et $b$ est la distance $AB$ et on a : $AB = |a - b|$.
\end{Définition}

\begin{Proposition}
Soient $x$ et $y$ deux nombres réels. On a :
\begin{enumerate}
	\item $|x| \geq 0$
	\item $|-x| = |x|$
	\item $|x - y| = |y - x|$
	\item $|x^2| = |x|^2 = x^2$
	\item \textcolor{red}{$\sqrt{x^2} = |x|$}
	\item $|x\times y| = |x||y|$
	\item si $y\neq 0$, alors $\left|\dfrac{x}{y}\right| = \dfrac{|x|}{|y|}$
	\item \textcolor{red}{$|x + y| \leq |x| + |y|$}
	\item $|x - y| \geq |x| - |y|$
	\item \textcolor{red}{$|x| = |y|$ signifie que $x = y$ ou $x = - y$}.
\end{enumerate}
\end{Proposition}

\textcolor{red}{Remarque : }
$$
\begin{cases}
|x| = r \\
r > 0
\end{cases} \text{signifie que } (x = r \text{ ou } x = -r)
$$

\begin{Exemple}
\begin{itemize}
	\item $|x| = 4$ signifie que ($x = 5$ ou $x = -5$).
	\item $|x| = -4$ est impossible car $|x| \geq 0$ et $-4 < 0$.
	\item $|x - 1| = 3$ signifie que ($x - 1 = 3$ ou $x - 1 = -3$) c'est-à-dire ($x = 4$ ou $x  = -2$)
\end{itemize}
\end{Exemple}

\begin{Application}
Soit $x;y\in\mathbb{R}$ tel que $x\leq 2$ et $y\geq 3$. Simplifier le nombre $A$ tel que $A = \sqrt{(x - 2)^2} + \sqrt{(y - 3)^3}$.
\end{Application}

\begin{Proposition}
Soit $x;y\in\mathbb{R}$ et $r > 0$.
\begin{enumerate}
	\item $|x| \leq r$ signifie que $-r\leq x\leq r$

\end{Proposition}

\mysection{Encadrement}
\begin{Définition}
Soient $a$ et $b$ deux nombres réels tels que $a<b$.
Toute double inégalité (parmi les doubles inégalités suivantes) $a<x<b\;\ a<x\leq b\;\ a\leq x < b\;\ a\leq x\leq b$ est appelée encadrement, de $x$, d'amplitude $b - a$.
\end{Définition}

\begin{Exemple}
\begin{itemize}
	\item $3,14 < x < 3,15$ est un encadrement de $\pi$ d'amplitude $3,15 - 3,14 = 0,01$.
	\item $1< \dfrac{3}{2} < 2$ est un encadrement de $\dfrac{3}{2}$ d'amplitude $2 - 1 = 1$.
\end{itemize}
\end{Exemple}

\begin{Application}
\begin{enumerate}
	\item Déterminer un encadrement de $\sqrt{5}$ d'amplitude $1$.
	\item Déterminer un encadrement de $\dfrac{9}{8}$ d'amplitude $2$ et un autre encadrement de $\dfrac{9}{8}$ d'amplitude $0,02$.
\end{enumerate}
\end{Application}

\mysubsection{1}{Encadrement de la somme de deux nombres}
\begin{Proposition}
Si $a\leq x\leq b$ et $c\leq x\leq d$ sont des encadrements des deux nombres réels $x$ et $y$, alors :
$\begin{cases}
a + c \leq x + y \leq b + d \\
a - d \leq x - y \leq b - c
\end{cases}$ son des encadrement des réels $x+y$ et $x-y$.
\end{Proposition}

\begin{Exemple}
\begin{itemize}
	\item On a : $1,41 < \sqrt{2} < 1,42$ et $1,732 < \sqrt{3} < 1,733$.
	
	Donc $3,142 < \sqrt{3} + x < 3,153$ est un encadrement de $\sqrt{3} + \sqrt{2}$ d'amplitude $3,153 - 3,142 = 0,011$.
	\item On a : $1,732 - 1,42 < \sqrt{3} - \sqrt{2} < 1,733 - 1,41$ c'est-à-dire : $0,312 < \sqrt{3} - \sqrt{2} < 0,323$ est un encadrement de $\sqrt{3} - \sqrt{2}$ d'amplitude de $0,01$.
\end{itemize}
\end{Exemple}
\mysubsection{2}{Encadrement du produit de deux nombres}
\begin{Proposition}
Soient $a,b,c$ et $d$ des nombres réels \underline{positifs}.

Si $a\leq x\leq b$ et $x\leq y\leq d$ sont deux encadrement de $x$ et $y$, alors $ax\leq xy\leq bd$ est encadrement de $xy$.
\end{Proposition}

\begin{Exemple}
\begin{itemize}
	\item Soit $1,3\leq x\leq 1,4$ et $3,9\leq y\leq 4$, alors $5,07\leq xy\leq 5,6$.
	\item Soit $-17,4\leq a\leq -14,3$ et $-22\leq b\leq -21,9$...
\end{itemize}
\end{Exemple}
\textcolor{red}{Remarque : }

Si $a\leq x\leq b$, alros $-b\leq -x \leq -a$.

\begin{Application}
Donner un encadrement de $x + y$ et $xy$ dans chacun des cas suivants :
\begin{enumerate}
	\item $1,01 \leq x\leq 1,02$ et $1,9\leq y\leq 2$
	\item $-2\leq x\leq -1$ et $4\leq y \leq 4,5$
	\item $-4\leq x\leq 2$ et $1\leq y\leq 3$
	\item $-1\leq x\leq 1$ et $1\leq y\leq 2$
\end{enumerate}
\end{Application}
\mysubsection{3}{Encadrement de l'inverse et du quotient}
\begin{Proposition}
Soient $a,b,c$ et $d$ des nombres réels \underline{strictement positifs}.

Si $a\leq x\leq b$ et $c\leq y\leq d$ sont deux encadrements des deux nombres réels $x$ et $y$, alors : $\dfrac{1}{b}\leq x\leq \dfrac{1}{a}$ et $\dfrac{1}{d}\leq y\leq \dfrac{1}{c}$ et $\dfrac{a}{d}\leq \dfrac{x}{y}\leq\dfrac{b}{c}$ sont deux encadrements des réels $\dfrac{1}{y}$ et $\dfrac{x}{y}$.
\end{Proposition}

\begin{Exemple}
\begin{itemize}
	\item Soit $2\leq x\leq 4$ et $1\leq y\leq 3$.

Alors $\dfrac{1}{3}\leq y\leq 1$(encadrement de $y$) et $\dfrac{2}{3}\leq \dfrac{x}{y}\leq 4$ (encadrement de $\dfrac{x}{y}$.
	\item $-4\leq a\leq -2$ et $1\leq b\leq 3$.
\end{itemize}
\end{Exemple}

\begin{Application}
\begin{enumerate}
	\item Soit $3,95\leq x\leq 4,01$ et $-2\leq y\leq -1,2$. Donner un encadrement de $\dfrac{x}{y}$
	\item Soit $0\leq a\leq 3$ et $-4\leq b\leq -2$. Donner un encadrement de chacun des deux nombres : $\dfrac{a}{b}$ et $\dfrac{2a + 3b}{3a - 2b}$.
\end{enumerate}
\end{Application}

\mysection{Approximation}

\mysubsection{1}{Approximation par défaut et par excès}
\begin{Définition}
Soit $a\leq x\leq b$ ($a < x < b$ ; $a < x \leq b$ ; $a\leq x < b$) un encadrement du nombre $x$ d'amplitude $b-a$.
\begin{itemize}
	\item Le nombre $a$ est appelé approximation du nombre $x$ à prés par défaut.
	\item Le nombre $b$ est appelé approximation du nombre $x$ à prés par excès.
\end{itemize}
\end{Définition}

\begin{Exemple}
\begin{itemize}
	\item On a $3,14 < \pi < 3,15$.
	
	Donc : $3,14$ est une approximation du nombre $\pi$ à $0,01$ prés par défaut
	
	$3,15$ est une approximation du nombre $\pi$ à $0,01$ prés par excès.
	\item On a $1,4142 < \sqrt{2} < 1,4143$.
	
	Donc : $1,4142$ est une approximation du nombre $\sqrt{2}$ à $0,0001 = 10^{-4}$ près par défaut
	
	$1,4143$ est une approximation du nombre $\sqrt{2}$ à $0,0001$ prés par excès.
\end{itemize}
\end{Exemple}

\begin{Application}
\begin{enumerate}
	\item Donner une approximation de $\dfrac{5}{4}$ à $10^{-1}$ près par excès.
	\item Donner une approximation de $\dfrac{}{25}$ à $10^{-2}$ près par défaut.	
\end{enumerate}
\end{Application}

\mysubsection{2}{Valeur approchée}
\begin{Définition}
Soient $x$ un nombre réel et $r$ un nombre réel strictement positif.

Tout nombre réel $a$ qui vérifie l'une ou l'autre des deux inégalités $|x - a| < r$ ou $|x - a|\leq r$ est appelé valeur approchée (ou approximation) du nombre $x$ à $r$ près (ou à la précision $r$ prés).
\end{Définition}

\begin{Exemple}
On a : $|\sqrt{5} - 2,235|\leq 0,005$. Donc $2,235$ est une valeur approchée du nombre $\sqrt{5}$ à la précision $0,005$ (ou $0,005$ près).
\end{Exemple}

\begin{Application}
\begin{enumerate}
	\item Déterminer une valeur approchée de $\dfrac{8}{7}$ à $10^{-2}$ près.
	\item Soit $a\leq x\leq b$ un encadrement du nombre $x$.
	
	Montrer que $b$ est une valeur approchée de $x$ à $\dfrac{b-a}{2}$ près. 
\end{enumerate}
\end{Application}

\mysection{Approximation décimales}
\begin{Définition}
Soit $x$ un nombre réel tel que : $N\times 10^{-p}\leq x\leq (N+1)\times 10^{-p}$ où $p\in\mathbb{N}$ et $N\in\mathbb{Z}$.
\begin{itemize}
	\item Le nombre $N\times 10^{-p}$ est appelé l'approximation décimales du nombre $x$ à $10^{-p}$ prés par défaut.
	\item Le nombre $(N+1)\times 10^{-p}$ est appelé l'approximation décimales du nombre $x$ à $10^{-p}$ prés par excès.
\end{itemize}
\end{Définition}

\begin{Exemple}
On a : $0,62<\dfrac{5}{8} < 0,63$ c'est-à-dire $62\times 10^{-2} < \dfrac{5}{8} < (62 + 1)\times 10^{-2}$.

Donc : $62\times 10^{-2} = 0,62$ est l'approximation décimale du nombre $\dfrac{5}{8}$ à $10^{-2}$ près par défaut.

$63\times 10^{-2} = 0,63$ est l'approximation décimale du nombre $\dfrac{5}{8}$ à $10^{-2}$ près par excès.
\end{Exemple}
\begin{Application}
\begin{enumerate}
	\item Donner l'approximation décimale du nombre $\sqrt{3}$ à $10^{-2}$ prés par défaut.
	\item Donner l'approximation décimale du nombre $\sqrt{2}$ à $10^{-2}$ prés par excès.
	\item On pose $a = \dfrac{\sqrt{7} - \sqrt{3}}{2}$
	
	Sachant que $1,73 < \sqrt{3} < 1,74$ et $2,64 < \sqrt{7} < 2,65$
	
	Donner les approximations décimales à $10^{-2}$ prés par défaut et par excès du nombre $a$.
\end{enumerate}
\end{Application}
