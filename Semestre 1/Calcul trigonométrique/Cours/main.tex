\begin{boxone}
{\Large\ding{45}}
\textcolor{red}{\large Contenus du programme :}
\begin{itemize}
   \item Cercle trigonométrique, les abscisses curvilignes d’un point, l’abscisse curviligne principale.
    \item Angle orienté de deux demi-droites ayant même origine, la mesure principale, relation de Chasles.
    \item Relation entre le degré, le radian et le grade.
    \item Angle orienté de deux vecteurs et mesure de cet angle.
    \item Lignes trigonométriques d’un nombre réel et lignes trigonométriques d’un angle de deux vecteurs.
    \item Relations :
        \begin{itemize}
            \item \( \cos^2 x + \sin^2 x = 1 \)
            \item \( \tan x = \dfrac{\sin x}{\cos x} \)
            \item \( \dfrac{1}{\cos^2 x} = 1 + \tan^2 x \)
        \end{itemize}
\end{itemize}

{\Large\ding{45}}
\textcolor{red}{\large Les capacités attendues :}
\begin{itemize}
    \item Utiliser la calculatrice scientifique pour déterminer une valeur approchée d'un angle défini par l'un de ses lignes trigonométriques et inversement.
    \item Maitriser les lignes trigonométriques des angles usuels et appliquer les différentes relations.
  \end{itemize}

{\Large\ding{45}}
\textcolor{red}{\large Recommandations pédagogiques :}
\begin{itemize}
	 \item On définira tout point du cercle trigonométrique par son abscisse curviligne principale ou par ses coordonnées par rapport à un repère orthonormé lié au cercle trigonométrique.
\end{itemize}
\end{boxone}
